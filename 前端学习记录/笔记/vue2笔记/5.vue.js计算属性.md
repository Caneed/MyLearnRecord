### computed

使用`computed`计算属性

反转字符串

```
<div id="app">
  {{ message.split('').reverse().join('') }}
</div>
```

使用`computed`

```
<body>
  <div id="app">
    <p>{{message}}</p>
    <p>{{reverseMessage}}</p>
  </div>
</body>
<script>
 var vm=new Vue({
  el:"#app",
  data:{
    message:"fuck you"
  },
  computed:{
    reverseMessage:function(){
      //this指向vm实例
      return this.message.split('').reverse().join('')
    }
  }
 })
</script>
```
