### v-if

`v-if`可以切换元素是否可见

```
<body>
  <div id="SeenMe">
    <span v-if="seen">now you see me</span>
  </div>
</body>
<script>
  const SeenMe={
    data(){
      return{
        seen:true
      }
    }
  }
  Vue.createApp(SeenMe).mount("#SeenMe")
</script>
```

### v-for

`v-for`指令可以绑定数组数据渲染项目列表

```
//html
 <div id="list">
    <ul>
      <li v-for="item in items">
        {{item.text}}
      </li>
    </ul>
  </div>
//js
 const List={
    data(){
      return{
        items:[
        { text: 'Learn JavaScript' },
        { text: 'Learn Vue' },
        { text: 'Build something awesome' }
        ]
      }
    }
  }
  Vue.createApp(List).mount("#list")
```
